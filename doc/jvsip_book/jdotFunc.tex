\afuncT{jdot}{Complex Vector Conjugate Dot Product}{matrixOperations}
\\ \hspace*{.8cm}For complex vector views $a$ and $b$
\begin{equation*}
\alpha = \sum \limits_{i=0}^{N-1} a_i \cdot \opConj(b_i)
\end{equation*}
