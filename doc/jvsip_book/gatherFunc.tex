\pyComment{
\item{Method \ttbf{gather} will create an output view if none is supplied.}
\item{Function \ttbf{gather} requires an output view be supplied.}
\item{Argument \ttbf{out} has the same precision as \ttbf{in}.}
\item{Argument \ttbf{out} has shape vector.}
\item{Argument \ttbf{in} has shape matrix or vector.}
\item{Argument \ttbf{indx} has shape vector of precision \ttbf{vi} or \ttbf{mi} depending on the shape of \ttbf{in}.}
\item{The length of \ttbf{indx} must be the same as the length of \ttbf{out}.}
