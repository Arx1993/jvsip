% View Methods
\\\hspace*{.8cm}{\textbf{View Methods\vspace{.2cm}}\\
\hspace*{1cm}\parbox{10.5cm}{
\begin{itemize}
\item {Special \ttbf{view} methods exist for \ttbf{view}s of type float and double.} 
\item {\ttbf{View} methods are defined as properties (@property). The scale factor is always one for \ttbf{view} methods.}
\item {For out of place the method will create and return the output \ttbf{view}.}
\item {\ttbf{View} methods determine if the FFT is a multiple FFT or a vector FFT by the \ttbf{view} type. The \pyjv \ttbf{view} \ttbf{major} attribute is used to determine if the multiple FFT is by row or by column.}
\item {Out-of-place \ttbf{view} methods \ttbf{fftop} and \ttbf{ifftop} treat real vectors as if they were complex with a zero imaginary part}
\end{itemize}}\\
%%
\hspace*{1.1cm}\textbf{In-Place: }\hspace{.2cm} yes\\
\hspace*{1.1cm}\textbf{Example: }\\
\hspace*{1.4cm}\parbox[t]{.8\textwidth}{Forward transform of vector \ttbf{x} in-place\\
\hspace*{.5cm}\ttbf{x.fftip}}\\\\
\hspace*{1.4cm}\parbox[t]{.8\textwidth}{For matrix FFT multiple use major attribute.\\
\hspace*{.5cm}\ttbf{x.ROW.fftip}\\
\hspace*{.5cm}\ttbf{x.COL.fftip}}\\\\
\hspace*{1.4cm}\parbox[t]{.8\textwidth}{Inverse transform of vector \ttbf{x} in-place\\
\hspace*{.5cm}\ttbf{x.ifftip}}\\\\
%%
\hspace*{1.1cm}\textbf{Out-of-Place: }\hspace{.2cm} yes\\
\hspace*{1.1cm}\textbf{Example: }\\
\hspace*{1.4cm}\parbox[t]{.8\textwidth}{Real to complex and complex to real FFT\\
\hspace*{.5cm}\ttbf{ y=x.rcfft}\\
\hspace*{.5cm}\ttbf{ z=y.crfft}}\\\\
\hspace*{1.4cm}\parbox[t]{.8\textwidth}{Complex to complex transform of vector \ttbf{x} out-of-place\\
\hspace*{.5cm}\ttbf{ y=x.fftop}}\\\\
 \hspace*{1.4cm}\parbox[t]{.8\textwidth}{Complex to complex inverse transform of vector \ttbf{x} out-of-place\\
\hspace*{.5cm}\ttbf{ y=x.ifftop}}\\\\
 \hspace*{1.4cm}\parbox[t]{.8\textwidth}{Complex to complex multiple transform of matrix \ttbf{x} out-of-place by column\\
\hspace*{.5cm}\ttbf{ y=x.COL.fftop}}\\\\
 \hspace*{1.4cm}\parbox[t]{.8\textwidth}{Complex to complex multiple transform of matrix \ttbf{x} out-of-place by row\\
\hspace*{.5cm}\ttbf{ y=x.ROW.fftop}}\\\\
 \hspace*{1.4cm}\parbox[t]{.8\textwidth}{Complex to complex multiple inverse transform of matrix \ttbf{x} out-of-place by column\\
\hspace*{.5cm}\ttbf{ y=x.COL.ifftop}}\\\\
 \hspace*{1.4cm}\parbox[t]{.8\textwidth}{Complex to complex multiple inverse transform of matrix \ttbf{x} out-of-place by row\\
\hspace*{.5cm}\ttbf{ y=x.ROW.ifftop}}\\\\

\hspace*{.8cm}{\textbf{FFT Class\vspace{.2cm}}\\
\hspace*{1.cm}\parbox{.9\textwidth}{To create an FFT object use \\*
\hspace*{1.cm} \ttbf{fftObj=FFT(t,*args)}\\*
where \ttbf{args} is a variable arguments list containing the create parameters for the FFT type selected, and \ttbf{t} is a string indicating the type of FFT to create.\vspace{.2cm}}\\
\hspace*{1.cm}\parbox{.9\textwidth}{The \ttbf{args} will contain some or all of the following in the order listed. The exact argument list for each type is shown in the Discrete Fourier Transform Class table below.\\
\begin{tabular}[t]{|l l|}
\multicolumn{2}{c}{\parbox[t]{.68\textwidth}{\center{\rmfamily \bfseries Fast Fourier Transform Argument List\vspace{.2cm}}}}\\ \hline
\ttbf{M} & \parbox[t]{.70\textwidth}{Column Length \vspace*{.1cm}}\\ \hline
\ttbf{N} & \parbox[t]{.70\textwidth}{Row Length for \ttbf{matrix} or Vector length for \ttbf{vector}\vspace*{.1cm}} \\\hline
\ttbf{scl} & \parbox[t]{.70\textwidth}{Scale Factor \vspace*{.1cm}}\\\hline
\ttbf{dir} & \parbox[t]{.70\textwidth}{Direction flag for FFT either VSIP\_FFT\_FWD or VSIP\_FFT\_INV \vspace*{.1cm}}\\\hline
\ttbf{major} & \parbox[t]{.70\textwidth}{For multiple FFT by row (VSIP\_ROW) or by column (VSIP\_COL) \vspace*{.1cm}}\\\hline
\ttbf{ntimes} & \parbox[t]{.70\textwidth}{Hint for how much the FFT object will be used. Zero indicates many times\vspace*{.1cm}} \\\hline
\ttbf{alghint} & \parbox[t]{.70\textwidth}{Algorithm hint to optimize for speed (VSIP\_ALG\_TIME), size (VSIP\_ALG\_SPACE), or accuracy (VSIP\_ALG\_NOISE)\vspace*{.1cm}}\\
